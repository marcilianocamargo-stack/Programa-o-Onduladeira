function buscarDadosOF() { const numOF = $('fOF').value.trim(); if (!numOF) { alert('Digite o número da OF para buscar.'); return; } // Busca no banco de dados local (localStorage) const o = db.ofs.find(x => x.of === numOF); if (!o) { alert('OF não encontrada no histórico local.'); return; } // Preenche os campos do formulário com os dados encontrados $('fCliente').value = o.cliente; $('fOC').value = o.oc || ''; $('fProduto').value = o.produto + '|' + o.onda; $('fQtde').value = o.qtde; $('fLarg').value = o.larg; $('fCompr').value = o.compr; $('fVinco').value = o.vinco; // Atualiza a estrutura de papéis do produto selecionado aoMudarProduto(); // Força o preenchimento dos papéis salvos (o.papeis || []).forEach((c, i) => { if ($('fPapel' + i) && papel(c)) $('fPapel' + i).value = c; }); $('fArranjo').value = o.arranjo; $('fBobina').value = o.bobina; $('fPalQtde').value = o.palQtde || ''; $('fPalMed').value = o.palMed || ''; // Recalcula as regras de negócio, régua e paletização na tela recalcula(); alert('Dados da OF ' + numOF + ' carregados com sucesso.'); }
